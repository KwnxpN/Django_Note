from datetime import datetime, timedelta

today = datetime.now()
print(f"datetime.now = {today}")
future_date = today + timedelta(days=500)
day_of_week_index = future_date.weekday()
days_of_week_th = ["วันจันทร์", "วันอังคาร", "วันพุธ", "วันพฤหัสบดี", "วันศุกร์", "วันเสาร์", "วันอาทิตย์"]
day_name = days_of_week_th[day_of_week_index]

print(f"นับจากวันนี้ไปอีก 500 วันจะเป็นวันที่: {future_date}")
print(f"ซึ่งตรงกับ: {day_name}")
